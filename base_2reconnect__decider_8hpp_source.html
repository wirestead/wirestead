42inline std::chrono::milliseconds clamp_reconnect_delay(std::chrono::milliseconds delay) {
43 if (delay < std::chrono::milliseconds(0)) {
44 return std::chrono::milliseconds(0);
45 }
46 return std::min(delay, MAX_RECONNECT_DELAY);
47}

55template <typename ConfigType>
56inline ReconnectLogicDecision decide_reconnect_common(const ConfigType& cfg, const diagnostics::ErrorInfo& error_info,
57 uint32_t attempt_count,
58 const std::optional<ReconnectPolicy>& policy) {
59 if (!error_info.retryable) {
60 return {false, std::nullopt};
61 }
62
63 if (cfg.max_retries == 0) {
64 return {false, std::nullopt};
65 }
66
67 if (cfg.max_retries > 0 && attempt_count >= static_cast<uint32_t>(cfg.max_retries)) {
68 return {false, std::nullopt};
69 }
70
71 if (policy) {
72 auto policy_decision = (*policy)(error_info, attempt_count);
73 if (!policy_decision.retry) {
74 return {false, std::nullopt};
75 }
76 return {true, clamp_reconnect_delay(policy_decision.delay)};
77 }
78
79 return {true, clamp_reconnect_delay(std::chrono::milliseconds(cfg.retry_interval_ms))};
80}
