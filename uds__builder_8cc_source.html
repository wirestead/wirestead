44template <uint32_t State>
45std::unique_ptr<wrapper::UdsClient> UdsClientBuilder<State>::build() {
46 std::unique_ptr<wrapper::UdsClient> client;
47 if (independent_context_) {
48 client = std::make_unique<wrapper::UdsClient>(socket_path_, std::make_shared<boost::asio::io_context>());
49 client->manage_external_context(true);
50 } else {
51 client = std::make_unique<wrapper::UdsClient>(socket_path_);
52 }
53
54 if (this->on_data_) client->on_data(this->on_data_);
55 if (this->on_data_batch_) client->on_data_batch(this->on_data_batch_);
56 if (this->on_connect_) client->on_connect(this->on_connect_);
57 if (this->on_disconnect_) client->on_disconnect(this->on_disconnect_);
58 if (this->on_error_) client->on_error(this->on_error_);
59 if (this->on_backpressure_) client->on_backpressure(this->on_backpressure_);
60
61 if (retry_interval_set_) client->retry_interval(retry_interval_);
62 if (max_retries_set_) client->max_retries(max_retries_);
63 if (connection_timeout_set_) client->connection_timeout(connection_timeout_);
64
65 if (this->bp_strategy_set_) client->backpressure_strategy(this->bp_strategy_);
66 client->backpressure_threshold(this->get_effective_backpressure_threshold());
67
68 if (this->framer_factory_) {
69 client->framer(this->framer_factory_());
70 }
71 if (this->on_message_) {
72 client->on_message(std::move(this->on_message_));
73 }
74 if (this->on_message_batch_) {
75 client->on_message_batch(std::move(this->on_message_batch_));
76 }
77
78 if (auto_start_) {
79 client->auto_start(true);
80 }
81
82 return client;
83}
