68inline std::string ts_now() {
69 using namespace std::chrono;
70 const auto now = system_clock::now();
71 const auto tt = system_clock::to_time_t(now);
72 std::tm tm{};
73#if defined(UNILINK_PLATFORM_WINDOWS)
74 localtime_s(&tm, &tt);
75#else
76 localtime_r(&tt, &tm);
77#endif
78 const auto ms = duration_cast<milliseconds>(now.time_since_epoch()) % 1000;
79 std::ostringstream oss;
80 oss << std::put_time(&tm, "%F %T") << '.' << std::setw(3) << std::setfill('0') << ms.count();
81 return oss.str(); // e.g., 2025-09-15 13:07:42.123
82}

107// Safe memory operations
108namespace safe_memory {
116inline void safe_memcpy(uint8_t* dest, const uint8_t* src, size_t size) {
117 if (!dest) {
118 throw std::invalid_argument("Destination pointer is null");
119 }
120 if (!src) {
121 throw std::invalid_argument("Source pointer is null");
122 }
123 if (size == 0) {
124 return; // Nothing to copy
125 }
126 if (size > constants::MAX_BUFFER_SIZE) {
127 throw std::invalid_argument("Copy size too large (max MAX_BUFFER_SIZE)");
128 }
129
130 std::memcpy(dest, src, size);
131}

142}
