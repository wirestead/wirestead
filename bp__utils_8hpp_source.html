32struct DropAccounting {

35
36 bool any() const { return messages > 0 || bytes > 0; }

39inline bool try_reserve_write_bytes(std::atomic<size_t>& queue_bytes, const std::atomic<size_t>& pending_bytes,
40 const std::atomic<bool>& backpressure_active, size_t bytes, size_t bp_high,
41 size_t bp_limit) {
42 if (bytes == 0 || bytes > bp_limit || backpressure_active.load(std::memory_order_relaxed)) return false;
43
44 size_t current = queue_bytes.load(std::memory_order_relaxed);
45 for (;;) {
46 const size_t pending = pending_bytes.load(std::memory_order_relaxed);
47 if (current > bp_high || bytes > bp_high - current) return false;
48 if (current > bp_limit || pending > bp_limit - current || bytes > bp_limit - current - pending) return false;
49
50 if (queue_bytes.compare_exchange_weak(current, current + bytes, std::memory_order_acq_rel,
51 std::memory_order_relaxed)) {
52 return true;
53 }
54 }
55}
56
57inline void release_reserved_write_bytes(std::atomic<size_t>& queue_bytes, size_t bytes) {
58 size_t current = queue_bytes.load(std::memory_order_relaxed);
59 for (;;) {
60 const size_t next = current > bytes ? current - bytes : 0;
61 if (queue_bytes.compare_exchange_weak(current, next, std::memory_order_acq_rel, std::memory_order_relaxed)) return;
62 }
63}

65// Returns the byte size of a buffer held in a transport BufferVariant alternative.
66// shared_ptr<const vector<uint8_t>> goes through ->size(); everything else via .size().
67template <typename T>
68inline size_t variant_buffer_size(const T& buf) {
69 if constexpr (std::is_same_v<T, std::shared_ptr<const std::vector<uint8_t>>>) {
70 return buf ? buf->size() : 0;
71 } else {
72 return buf.size();
73 }
74}

76// Identity projection: the default for maybe_flush_for_keep_latest()'s `project`
77// parameter below, used as-is by transports whose tx_ deque holds the
78// BufferVariant directly. UDP's tx_ holds TxItem{BufferVariant, destination}
79// instead, and supplies a projection extracting `.buffer` so this same
80// trimming logic can still visit the variant inside.
81struct IdentityProjection {
82 template <typename T>
83 constexpr T& operator()(T& x) const {
84 return x;
85 }
86};
87
88// BestEffort queue-trimming shared by all stream transports (TCP client/server, UDS client/server)
89// and, via a projection, UDP.
90// Must be called on the strand immediately before enqueueing a new buffer of `added` bytes.
91//
92// No-op for Reliable strategy.
93// For BestEffort:
94// added >= bp_high → drop entire tx_ (full keep-latest replacement).
95// otherwise → pop oldest tx_ entries until queue_bytes + added <= bp_high.
96template <typename Deque, typename Project = IdentityProjection>
97inline DropAccounting maybe_flush_for_keep_latest(::unilink::base::constants::BackpressureStrategy bp_strategy,
98 size_t added, size_t bp_high, Deque& tx,
99 std::atomic<size_t>& queue_bytes,
100 const std::atomic<bool>& backpressure_active,
101 Project project = Project{}) {
102 DropAccounting dropped;
103 if (bp_strategy != ::unilink::base::constants::BackpressureStrategy::BestEffort) return dropped;
104
105 if (added >= bp_high) {
106 size_t removed_bytes = 0;
107 for (auto& buf : tx) {
108 removed_bytes += std::visit([](const auto& b) { return variant_buffer_size(b); }, project(buf));
109 }
110 dropped.messages = tx.size();
111 dropped.bytes = removed_bytes;
112 tx.clear();
113 const size_t qb = queue_bytes.load(std::memory_order_relaxed);
114 queue_bytes.store(qb > removed_bytes ? qb - removed_bytes : 0, std::memory_order_relaxed);
115 return dropped;
116 }
117
118 if (backpressure_active.load(std::memory_order_relaxed) ||
119 queue_bytes.load(std::memory_order_relaxed) + added > bp_high) {
120 while (!tx.empty()) {
121 const size_t qb = queue_bytes.load(std::memory_order_relaxed);
122 if (qb + added <= bp_high) break;
123 const size_t oldest = std::visit([](const auto& b) { return variant_buffer_size(b); }, project(tx.front()));
124 queue_bytes.store(qb > oldest ? qb - oldest : 0, std::memory_order_relaxed);
125 tx.pop_front();
126 ++dropped.messages;
127 dropped.bytes += oldest;
128 }
129 }
130 return dropped;
131}
