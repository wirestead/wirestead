140inline void InputValidator::validate_retry_count(int retry_count) {
141 if (retry_count == base::constants::DEFAULT_MAX_RETRIES) { // -1 means infinite, which is valid
142 return;
143 }
144 if (retry_count < FINITE_MIN_RETRY_COUNT || retry_count > FINITE_MAX_RETRY_COUNT) {
145 throw diagnostics::ValidationException("retry_count out of range", "retry_count",
146 std::to_string(FINITE_MIN_RETRY_COUNT) + " <= retry_count <= " +
147 std::to_string(FINITE_MAX_RETRY_COUNT) + " or -1 for infinite");
148 }
149}

173inline void InputValidator::validate_memory_alignment(const void* ptr, size_t alignment) {
174 if (ptr == nullptr) {
175 throw diagnostics::ValidationException("memory pointer cannot be null", "ptr", "non-null pointer");
176 }
177
178 uintptr_t address = reinterpret_cast<uintptr_t>(ptr);
179 if (address % alignment != 0) {
180 throw diagnostics::ValidationException("memory pointer not properly aligned", "ptr",
181 "aligned to " + std::to_string(alignment) + " bytes");
182 }
183}
