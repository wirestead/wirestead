28struct TcpClientConfig {
29 std::string host = "127.0.0.1";
30 uint16_t port = 9000;
31 unsigned retry_interval_ms = base::constants::DEFAULT_RETRY_INTERVAL_MS;
32 unsigned connection_timeout_ms = base::constants::DEFAULT_CONNECTION_TIMEOUT_MS;
33 unsigned idle_timeout_ms = 0; // Idle connection timeout in milliseconds (0 = disabled)
34 IdleTimeoutAction idle_timeout_action = IdleTimeoutAction::Reconnect;
35 int max_retries = base::constants::DEFAULT_MAX_RETRIES;
36 size_t backpressure_threshold = base::constants::DEFAULT_BACKPRESSURE_THRESHOLD;
37 base::constants::BackpressureStrategy backpressure_strategy = base::constants::BackpressureStrategy::Reliable;
38 bool enable_memory_pool = true;
39 bool tcp_no_delay = true;
40 bool keep_alive = false;
41 size_t send_buffer_size = 0;
42 size_t receive_buffer_size = 0;
43
44 TcpClientConfig() = default;
45 TcpClientConfig(const TcpClientConfig&) = default;
46 TcpClientConfig& operator=(const TcpClientConfig&) = default;
47 TcpClientConfig(TcpClientConfig&&) noexcept = default;
48 TcpClientConfig& operator=(TcpClientConfig&&) noexcept = default;
49
50 // Validation methods
51 bool is_valid() const {
52 return util::InputValidator::is_valid_host(host) && port > 0 &&
53 retry_interval_ms >= base::constants::MIN_RETRY_INTERVAL_MS &&
54 retry_interval_ms <= base::constants::MAX_RETRY_INTERVAL_MS &&
55 backpressure_threshold >= base::constants::MIN_BACKPRESSURE_THRESHOLD &&
56 backpressure_threshold <= base::constants::MAX_BACKPRESSURE_THRESHOLD &&
57 (send_buffer_size == 0 || (send_buffer_size >= base::constants::MIN_SOCKET_BUFFER_SIZE &&
58 send_buffer_size <= base::constants::MAX_SOCKET_BUFFER_SIZE)) &&
59 (receive_buffer_size == 0 || (receive_buffer_size >= base::constants::MIN_SOCKET_BUFFER_SIZE &&
60 receive_buffer_size <= base::constants::MAX_SOCKET_BUFFER_SIZE)) &&
61 (max_retries == -1 || (max_retries >= 0 && max_retries <= base::constants::MAX_RETRIES_LIMIT));
62 }
63
64 // Apply validation and clamp values to valid ranges
65 void validate_and_clamp() {
66 if (retry_interval_ms < base::constants::MIN_RETRY_INTERVAL_MS) {
67 retry_interval_ms = base::constants::MIN_RETRY_INTERVAL_MS;
68 } else if (retry_interval_ms > base::constants::MAX_RETRY_INTERVAL_MS) {
69 retry_interval_ms = base::constants::MAX_RETRY_INTERVAL_MS;
70 }
71
72 if (backpressure_threshold < base::constants::MIN_BACKPRESSURE_THRESHOLD) {
73 backpressure_threshold = base::constants::MIN_BACKPRESSURE_THRESHOLD;
74 } else if (backpressure_threshold > base::constants::MAX_BACKPRESSURE_THRESHOLD) {
75 backpressure_threshold = base::constants::MAX_BACKPRESSURE_THRESHOLD;
76 }
77
78 if (max_retries != -1 && max_retries > base::constants::MAX_RETRIES_LIMIT) {
79 max_retries = base::constants::MAX_RETRIES_LIMIT;
80 }
81
82 if (send_buffer_size != 0 && send_buffer_size < base::constants::MIN_SOCKET_BUFFER_SIZE) {
83 send_buffer_size = base::constants::MIN_SOCKET_BUFFER_SIZE;
84 } else if (send_buffer_size > base::constants::MAX_SOCKET_BUFFER_SIZE) {
85 send_buffer_size = base::constants::MAX_SOCKET_BUFFER_SIZE;
86 }
87
88 if (receive_buffer_size != 0 && receive_buffer_size < base::constants::MIN_SOCKET_BUFFER_SIZE) {
89 receive_buffer_size = base::constants::MIN_SOCKET_BUFFER_SIZE;
90 } else if (receive_buffer_size > base::constants::MAX_SOCKET_BUFFER_SIZE) {
91 receive_buffer_size = base::constants::MAX_SOCKET_BUFFER_SIZE;
92 }
93 }
94};
