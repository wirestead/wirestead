150 void fulfill_all_locked(bool value) {
151 for (auto& p : pending_promises_) {
152 try {
153 p.set_value(value);
154 } catch (...) {
155 }
156 }
157 pending_promises_.clear();
158 }
159
160 void flush_batches() {
161 std::unique_lock<std::shared_mutex> lock(mutex_);
162 if (!data_batch_queue_.empty()) {
163 auto handler = data_batch_handler_;
164 auto batch = std::move(data_batch_queue_);
165 data_batch_queue_.clear();
166 if (handler) {
167 lock.unlock();
168 handler(batch);
169 lock.lock();
170 }
171 }
172 if (!message_batch_queue_.empty()) {
173 auto handler = message_batch_handler_;
174 auto batch = std::move(message_batch_queue_);
175 message_batch_queue_.clear();
176 if (handler) {
177 lock.unlock();
178 handler(batch);
179 lock.lock();
180 }
181 }
182 if (batch_timer_) {
183 batch_timer_->cancel();
184 }
185 }
186
187 void schedule_batch_timer() {
188 if (!batch_timer_) return;
189 batch_timer_->expires_after(max_batch_latency_);
190 batch_timer_->async_wait(
191 [this, weak_alive = std::weak_ptr<bool>(alive_marker_)](const boost::system::error_code& ec) {
192 if (ec) return;
193 auto alive = weak_alive.lock();
194 if (!alive) return;
195 flush_batches();
196 });
197 }

199 std::future<bool> start() {
200 std::unique_lock<std::shared_mutex> lock(mutex_);
201 if (is_listening_.load()) {
202 std::promise<bool> p;
203 p.set_value(true);
204 return p.get_future();
205 }
206 std::promise<bool> p;
207 auto f = p.get_future();
208 pending_promises_.push_back(std::move(p));
209 if (started_.exchange(true)) return f;
210
211 if (!channel_) {
212 config::TcpServerConfig config;
213 config.bind_address = bind_address_;
214 config.port = port_;
215 config.enable_port_retry = port_retry_enabled_.load();
216 config.max_port_retries = max_port_retries_.load();
217 config.port_retry_interval_ms = port_retry_interval_ms_.load();
218 config.idle_timeout_ms = idle_timeout_ms_.load();
219 config.backpressure_threshold = backpressure_threshold_.load();
220 config.backpressure_strategy = backpressure_strategy_.load();
221 config.tcp_no_delay = tcp_no_delay_.load();
222 config.keep_alive = keep_alive_.load();
223 config.send_buffer_size = send_buffer_size_.load();
224 config.receive_buffer_size = receive_buffer_size_.load();
225
226 channel_ = factory::ChannelFactory::create(config, external_ioc_);
227 transport_cache_ = std::dynamic_pointer_cast<transport::TcpServer>(channel_);
228 setup_internal_handlers();
229
230 if (client_limit_enabled_.load()) {
231 auto transport_server = std::dynamic_pointer_cast<transport::TcpServer>(channel_);
232 if (transport_server) {
233 transport_server->set_client_limit(max_clients_.load());
234 }
235 }
236 }
237 lock.unlock();
238 channel_->start();
239 if (use_external_context_.load() && manage_external_context_.load() && !external_thread_.joinable()) {
240 if (external_ioc_ && external_ioc_->stopped()) {
241 external_ioc_->restart();
242 }
243 work_guard_ = std::make_unique<boost::asio::executor_work_guard<boost::asio::io_context::executor_type>>(
244 boost::asio::make_work_guard(*external_ioc_));
245 external_thread_ = std::thread([ioc = external_ioc_]() {
246 try {
247 ioc->run();
248 } catch (...) {
249 }
250 });
251 }
252 return f;
253 }

255 void stop() {
256 bool should_join = false;
257 {
258 std::unique_lock<std::shared_mutex> lock(mutex_);
259 if (!started_.exchange(false)) {
260 is_listening_.store(false);
261 fulfill_all_locked(false);
262 return;
263 }
264 bp_cv_.notify_all();
265 if (batch_timer_) {
266 batch_timer_->cancel();
267 batch_timer_.reset();
268 }
269 if (channel_) {
270 channel_->on_bytes(nullptr);
271 channel_->on_state(nullptr);
272 auto transport_server = std::dynamic_pointer_cast<transport::TcpServer>(channel_);
273 if (transport_server) transport_server->request_stop();
274 lock.unlock();
275 channel_->stop();
276 lock.lock();
277 }
278 if (use_external_context_.load() && manage_external_context_.load()) {
279 if (work_guard_) work_guard_.reset();
280 if (external_ioc_) external_ioc_->stop();
281 should_join = true;
282 }
283 fulfill_all_locked(false);
284 is_listening_.store(false);
285 }
286 if (should_join && external_thread_.joinable()) {
287 try {
288 if (std::this_thread::get_id() != external_thread_.get_id()) {
289 external_thread_.join();
290 } else {
291 external_thread_.detach();
292 }
293 } catch (...) {
294 }
295 }
296 std::unique_lock<std::shared_mutex> lock(mutex_);
297 channel_.reset();
298 }

318 bool broadcast(std::string_view data) {
319 // In order to avoid Head-Of-Line blocking where a single slow client
320 // blocks the entire broadcast loop, we delegate to try_broadcast (async fan-out)
321 // even in Reliable mode. Transport-level backpressure will still protect the queues.
322 return try_broadcast(data);
323 }
324
325 // channel_->on_backpressure()/session-level backpressure callbacks call bp_cv_.notify_all()
326 // from the transport's io_context thread without holding bp_mutex_ - a classic lost-wakeup
327 // race is possible: a waiter can check the predicate, find it still blocking, and be in the
328 // process of registering to wait when the notify fires. Poll with a bounded timeout instead
329 // of an unbounded wait() so a missed notify only costs a short delay rather than a
330 // permanent hang (see #427, #431).
331 bool send_to_blocking(ClientId client_id, std::string_view data) {
332 std::unique_lock<std::mutex> lock(bp_mutex_);
333 auto predicate = [this, client_id]() {
334 std::shared_lock<std::shared_mutex> rlock(mutex_);
335 const auto& ts = transport_cache_;
336 return !started_.load() || !ts || !ts->is_backpressure_active(client_id);
337 };
338 while (!bp_cv_.wait_for(lock, std::chrono::milliseconds(50), predicate)) {
339 }
340 std::shared_lock<std::shared_mutex> rlock(mutex_);
341 const auto& ts = transport_cache_;
342 return ts ? ts->send_to_client(client_id, data) : false;
343 }

345 void setup_internal_handlers() {
346 if (!channel_) return;
347
348 batch_timer_ = std::make_unique<boost::asio::steady_timer>(channel_->get_executor());
349
350 std::weak_ptr<bool> weak_alive = alive_marker_;
351 auto transport_server = std::dynamic_pointer_cast<transport::TcpServer>(channel_);
352 if (transport_server) {
353 transport_server->on_multi_connect([this, weak_alive](ClientId id, const std::string& info) {
354 auto alive = weak_alive.lock();
355 if (!alive) return;
356
357 ConnectionHandler handler;
358 {
359 std::unique_lock<std::shared_mutex> lock(mutex_);
360 if (framer_factory_) {
361 auto framer = framer_factory_();
362 if (framer) {
363 auto shared_framer = std::shared_ptr<framer::IFramer>(std::move(framer));
364 shared_framer->on_message([this, id](memory::ConstByteSpan msg) {
365 std::unique_lock<std::shared_mutex> lock(mutex_);
366 if (message_batch_handler_) {
367 message_batch_queue_.emplace_back(id, memory::SafeDataBuffer(msg));
368 if (message_batch_queue_.size() >= max_batch_size_) {
369 auto handler = message_batch_handler_;
370 auto batch = std::move(message_batch_queue_);
371 message_batch_queue_.clear();
372 lock.unlock();
373 handler(batch);
374 } else if (message_batch_queue_.size() == 1) {
375 schedule_batch_timer();
376 }
377 return;
378 }
379
380 MessageHandler on_message_handler = on_message_;
381 if (on_message_handler) {
382 lock.unlock();
383 on_message_handler(MessageContext(id, memory::SafeDataBuffer(msg)));
384 }
385 });
386 framers_[id] = std::move(shared_framer);
387 }
388 }
389 handler = on_client_connect_;
390 }
391 if (handler) handler(ConnectionContext(id, info));
392 });
393 transport_server->on_multi_data([this, weak_alive](ClientId id, memory::ConstByteSpan data_span) {
394 auto alive = weak_alive.lock();
395 if (!alive) return;
396
397 std::shared_ptr<framer::IFramer> framer;
398 std::unique_lock<std::shared_mutex> lock(mutex_);
399 if (data_batch_handler_) {
400 data_batch_queue_.emplace_back(id, memory::SafeDataBuffer(data_span));
401 if (data_batch_queue_.size() >= max_batch_size_) {
402 auto handler = data_batch_handler_;
403 auto batch = std::move(data_batch_queue_);
404 data_batch_queue_.clear();
405 lock.unlock();
406 handler(batch);
407 lock.lock();
408 } else if (data_batch_queue_.size() == 1) {
409 schedule_batch_timer();
410 }
411 } else {
412 MessageHandler handler = on_data_;
413 if (handler) {
414 lock.unlock();
415 handler(MessageContext(id, memory::SafeDataBuffer(data_span)));
416 lock.lock();
417 }
418 }
419
420 auto it = framers_.find(id);
421 if (it != framers_.end()) {
422 framer = it->second;
423 }
424 lock.unlock();
425 if (framer) framer->push_bytes(data_span);
426 });
427 transport_server->on_multi_disconnect([this, weak_alive](ClientId id) {
428 auto alive = weak_alive.lock();
429 if (!alive) return;
430
431 ConnectionHandler handler;
432 {
433 std::unique_lock<std::shared_mutex> lock(mutex_);
434 framers_.erase(id);
435 handler = on_disconnect_;
436 }
437 if (handler) handler(ConnectionContext(id));
438 });
439
440 transport_server->on_backpressure([this, weak_alive](size_t queued) {
441 bp_cv_.notify_all();
442 auto alive = weak_alive.lock();
443 if (!alive) return;
444 std::function<void(size_t)> handler;
445 {
446 std::shared_lock<std::shared_mutex> lock(mutex_);
447 handler = on_backpressure_;
448 }
449 if (handler) handler(queued);
450 });
451 }
452 channel_->on_state([this, weak_alive](base::LinkState state) {
453 auto alive = weak_alive.lock();
454 if (!alive) return;
455
456 if (state == base::LinkState::Listening) {
457 is_listening_.store(true);
458 std::unique_lock<std::shared_mutex> lock(mutex_);
459 fulfill_all_locked(true);
460 } else if (state == base::LinkState::Error || state == base::LinkState::Closed ||
461 state == base::LinkState::Idle) {
462 ErrorHandler handler;
463 is_listening_.store(false);
464 {
465 std::unique_lock<std::shared_mutex> lock(mutex_);
466 fulfill_all_locked(false);
467 if (state == base::LinkState::Error) {
468 handler = on_error_;
469 }
470 }
471 if (handler) {
472 handler(ErrorContext(ErrorCode::IoError, "Server error"));
473 }
474 }
475 });
476 }
