48template <uint32_t State>
49std::unique_ptr<wrapper::Serial> SerialBuilder<State>::build() {
50 std::unique_ptr<wrapper::Serial> serial;
51 if (independent_context_) {
52 serial = std::make_unique<wrapper::Serial>(device_, baud_rate_, std::make_shared<boost::asio::io_context>());
53 serial->manage_external_context(true);
54 } else {
55 serial = std::make_unique<wrapper::Serial>(device_, baud_rate_);
56 }
57
58 if (this->on_data_) serial->on_data(this->on_data_);
59 if (this->on_data_batch_) serial->on_data_batch(this->on_data_batch_);
60 if (this->on_connect_) serial->on_connect(this->on_connect_);
61 if (this->on_disconnect_) serial->on_disconnect(this->on_disconnect_);
62 if (this->on_error_) serial->on_error(this->on_error_);
63 if (this->on_backpressure_) serial->on_backpressure(this->on_backpressure_);
64
65 if (char_size_set_) serial->data_bits(static_cast<int>(char_size_));
66 if (stop_bits_set_) serial->stop_bits(static_cast<int>(stop_bits_));
67
68 // Note: wrapper::Serial setters use strings for enum types
69 if (parity_set_) {
70 std::string p_str = "none";
71 if (parity_ == config::SerialConfig::Parity::Even)
72 p_str = "even";
73 else if (parity_ == config::SerialConfig::Parity::Odd)
74 p_str = "odd";
75 serial->parity(p_str);
76 }
77
78 if (flow_set_) {
79 std::string f_str = "none";
80 if (flow_ == config::SerialConfig::Flow::Software)
81 f_str = "software";
82 else if (flow_ == config::SerialConfig::Flow::Hardware)
83 f_str = "hardware";
84 serial->flow_control(f_str);
85 }
86
87 if (reopen_on_error_set_) serial->reopen_on_error(reopen_on_error_);
88 if (retry_interval_set_) serial->retry_interval(std::chrono::milliseconds(retry_interval_ms_));
89
90 if (this->bp_strategy_set_) serial->backpressure_strategy(this->bp_strategy_);
91 serial->backpressure_threshold(this->get_effective_backpressure_threshold());
92
93 if (this->framer_factory_) {
94 serial->framer(this->framer_factory_());
95 }
96 if (this->on_message_) {
97 serial->on_message(std::move(this->on_message_));
98 }
99 if (this->on_message_batch_) {
100 serial->on_message_batch(std::move(this->on_message_batch_));
101 }
102
103 if (auto_start_) {
104 serial->auto_start(true);
105 }
106
107 return serial;
108}

137template <uint32_t State>
138SerialBuilder<State>& SerialBuilder<State>::parity(const std::string& p) {
139 std::string value = p;
140 std::transform(value.begin(), value.end(), value.begin(),
141 [](unsigned char c) { return static_cast<char>(std::tolower(c)); });
142
143 if (value == "even") {
144 parity_ = config::SerialConfig::Parity::Even;
145 } else if (value == "odd") {
146 parity_ = config::SerialConfig::Parity::Odd;
147 } else {
148 parity_ = config::SerialConfig::Parity::None;
149 }
150 parity_set_ = true;
151 return *this;
152}
