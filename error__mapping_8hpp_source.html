34inline ErrorCode to_unilink_error_code(const boost::system::error_code& ec) {
35 if (!ec) [[likely]] {
36 return ErrorCode::Success;
37 }
38
39 // Check specific boost error codes
40 if (ec == boost::asio::error::connection_refused) [[unlikely]] {
41 return ErrorCode::ConnectionRefused;
42 }
43 if (ec == boost::asio::error::timed_out) [[unlikely]] {
44 return ErrorCode::TimedOut;
45 }
46 if (ec == boost::asio::error::connection_reset) [[unlikely]] {
47 return ErrorCode::ConnectionReset;
48 }
49 if (ec == boost::asio::error::connection_aborted) [[unlikely]] {
50 return ErrorCode::ConnectionAborted;
51 }
52 if (ec == boost::asio::error::network_unreachable || ec == boost::asio::error::host_unreachable) [[unlikely]] {
53 return ErrorCode::NotConnected;
54 }
55 if (ec == boost::asio::error::already_connected) [[unlikely]] {
56 return ErrorCode::AlreadyConnected;
57 }
58 if (ec == boost::asio::error::address_in_use) [[unlikely]] {
59 return ErrorCode::PortInUse;
60 }
61 if (ec == boost::asio::error::access_denied) [[unlikely]] {
62 return ErrorCode::AccessDenied;
63 }
64
65 // Fallback to generic IoError for other network errors
66 return ErrorCode::IoError;
67}

72inline bool is_retryable_tcp_connect_error(const boost::system::error_code& ec) {
73 if (!ec) [[likely]]
74 return false;
75
76 // Connection refused is temporary (server might be starting up)
77 if (ec == boost::asio::error::connection_refused) return true;
78 // Timeouts are temporary network glitches
79 if (ec == boost::asio::error::timed_out) return true;
80 // Reset connection might be temporary
81 if (ec == boost::asio::error::connection_reset) return true;
82 // Unreachable might be temporary routing issue
83 if (ec == boost::asio::error::network_unreachable || ec == boost::asio::error::host_unreachable) return true;
84 // Resource temporarily unavailable
85 if (ec == boost::asio::error::try_again) return true;
86
87 // Aborted usually means stopped by user, not retryable automatically
88 if (ec == boost::asio::error::operation_aborted) return false;
89
90 // Unknown network errors are treated as transient to err on the side of resilience.
91 return true;
92}

97inline bool is_retryable_uds_connect_error(const boost::system::error_code& ec) {
98 if (!ec) return false;
99
100 // For UDS, "connection refused" often means the socket file exists but no one is listening.
101 // "no such file or directory" means the server hasn't created the socket yet.
102 // Both are retryable.
103 if (ec == boost::asio::error::connection_refused) return true;
104 if (ec == boost::system::errc::no_such_file_or_directory) return true;
105 if (ec == boost::asio::error::timed_out) return true;
106 if (ec == boost::asio::error::try_again) return true;
107
108 if (ec == boost::asio::error::operation_aborted) return false;
109
110 return true;
111}

116inline wrapper::ErrorContext to_error_context(const diagnostics::ErrorInfo& info,
117 std::optional<size_t> client_id = std::nullopt) {
118 ErrorCode code = ErrorCode::IoError;
119
120 if (info.boost_error) {
121 code = to_unilink_error_code(info.boost_error);
122 } else {
123 // Map category if boost error is not present
124 switch (info.category) {
125 case ErrorCategory::CONNECTION:
126 code = ErrorCode::NotConnected;
127 break;
128 case ErrorCategory::CONFIGURATION:
129 code = ErrorCode::InvalidConfiguration;
130 break;
131 case ErrorCategory::SYSTEM:
132 code = ErrorCode::InternalError;
133 break;
134 default:
135 code = ErrorCode::IoError;
136 break;
137 }
138 }
139
140 // Use the message from ErrorInfo
141 return wrapper::ErrorContext(code, info.message, client_id);
142}
