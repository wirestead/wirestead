210void UdsClient::start() {
211 auto current_state = impl_->state_.state();
212 if (current_state == LinkState::Connecting || current_state == LinkState::Connected) {
213 return;
214 }
215
216 impl_->recalculate_backpressure_bounds();
217 impl_->stop_requested_ = false;
218 impl_->stopping_ = false;
219 impl_->current_seq_++;
220 uint64_t seq = impl_->current_seq_.load();
221
222 if (impl_->owns_ioc_ && !impl_->ioc_thread_.joinable()) {
223 if (impl_->ioc_->stopped()) {
224 impl_->ioc_->restart();
225 }
226 impl_->work_guard_ =
227 std::make_unique<net::executor_work_guard<net::io_context::executor_type>>(net::make_work_guard(*impl_->ioc_));
228 impl_->ioc_thread_ = std::jthread([ioc = impl_->owned_ioc_](std::stop_token st) {
229 try {
230 std::stop_callback cb(st, [ioc] { ioc->stop(); });
231 ioc->run();
232 } catch (...) {
233 }
234 });
235 }
236
237 net::post(impl_->strand_, [self = shared_from_this(), seq]() {
238 self->impl_->transition_to(LinkState::Connecting);
239 self->impl_->do_connect(self, seq);
240 });
241}

243void UdsClient::stop() {
244 bool already_stopping = impl_->stopping_.exchange(true);
245 if (already_stopping) return;
246
247 impl_->stop_requested_ = true;
248 impl_->connected_ = false;
249 const auto seq = impl_->current_seq_.fetch_add(1) + 1;
250
251 // Release work guard and allow io_context to run out of work
252 if (impl_->ioc_) {
253 if (auto self = weak_from_this().lock()) {
254 net::post(impl_->strand_, [self, seq]() { self->impl_->perform_stop_cleanup(seq); });
255 } else {
256 impl_->perform_stop_cleanup(seq);
257 }
258 } else {
259 impl_->perform_stop_cleanup(seq);
260 }
261
262 if (impl_->owns_ioc_ && impl_->ioc_thread_.joinable()) {
263 if (std::this_thread::get_id() == impl_->ioc_thread_.get_id()) {
264 impl_->ioc_thread_.detach();
265 } else {
266 impl_->ioc_thread_.join();
267 }
268 }
269
270 // Transition to Idle state (lock-free or safe call)
271 impl_->state_.set_state(LinkState::Idle);
272}

293bool UdsClient::async_write_move(std::vector<uint8_t>&& data) {
294 if (!impl_->connected_.load() || impl_->stop_requested_.load()) {
295 impl_->stats_.record_failed_send();
296 return false;
297 }
298 if (data.empty()) {
299 impl_->stats_.record_failed_send();
300 return false;
301 }
302 if (impl_->queue_bytes_ + impl_->pending_bytes_ + data.size() > impl_->bp_limit_) {
303 impl_->stats_.record_failed_send();
304 return false;
305 }
306 impl_->stats_.record_accepted(data.size());
307 net::post(impl_->strand_, [this, self = shared_from_this(), data = std::move(data)]() mutable {
308 size_t added = data.size();
309 impl_->route_enqueued_buffer(self, BufferVariant{std::move(data)}, added);
310 });
311 return true;
312}

339bool UdsClient::async_try_write_move(std::vector<uint8_t>&& data) {
340 if (!impl_->connected_.load() || impl_->stop_requested_.load()) {
341 impl_->stats_.record_failed_send();
342 return false;
343 }
344 const auto added = data.size();
345 if (added == 0 || added > base::constants::MAX_BUFFER_SIZE) {
346 impl_->stats_.record_failed_send();
347 return false;
348 }
349 const auto reject_for_pressure = [this, added]() {
350 if (impl_->bp_strategy_ == base::constants::BackpressureStrategy::BestEffort) {
351 impl_->stats_.record_dropped(1, added);
352 } else {
353 impl_->stats_.record_failed_send();
354 }
355 };
356 if (impl_->backpressure_active_.load() || impl_->queue_bytes_ + added > impl_->bp_high_ ||
357 impl_->queue_bytes_ + impl_->pending_bytes_ + added > impl_->bp_limit_) {
358 reject_for_pressure();
359 return false;
360 }
361 if (!queue_util::try_reserve_write_bytes(impl_->queue_bytes_, impl_->pending_bytes_, impl_->backpressure_active_,
362 added, impl_->bp_high_, impl_->bp_limit_)) {
363 reject_for_pressure();
364 return false;
365 }
366 impl_->stats_.record_accepted(added);
367
368 net::post(impl_->strand_, [this, self = shared_from_this(), data = std::move(data), added]() mutable {
369 if (!impl_->connected_.load() || impl_->stop_requested_.load()) {
370 queue_util::release_reserved_write_bytes(impl_->queue_bytes_, added);
371 impl_->stats_.record_failed_send();
372 return;
373 }
374
375 impl_->tx_.emplace_back(std::move(data));
376 impl_->observe_queue();
377 impl_->report_backpressure(self, impl_->queue_bytes_);
378 if (!impl_->writing_) impl_->do_write(self, impl_->current_seq_.load());
379 });
380 return true;
381}

465void UdsClient::Impl::do_connect(std::shared_ptr<UdsClient> self, uint64_t seq) {
466 if (stop_requested_.load() || stopping_.load() || seq != current_seq_.load()) {
467 return;
468 }
469
470 if (!cfg_.is_valid()) {
471 self->impl_->record_error(diagnostics::ErrorLevel::ERROR, diagnostics::ErrorCategory::CONFIGURATION, "connect",
472 make_error_code(boost::system::errc::invalid_argument), "Invalid UDS socket path", false,
473 self->impl_->reconnect_attempt_count_);
474 self->impl_->transition_to(LinkState::Error);
475 return;
476 }
477
478 std::shared_ptr<uds::endpoint> endpoint;
479 try {
480 endpoint = std::make_shared<uds::endpoint>(cfg_.socket_path);
481 } catch (const std::exception& e) {
482 self->impl_->record_error(diagnostics::ErrorLevel::ERROR, diagnostics::ErrorCategory::CONFIGURATION, "connect",
483 make_error_code(boost::system::errc::filename_too_long),
484 "Invalid UDS endpoint: " + std::string(e.what()), false,
485 self->impl_->reconnect_attempt_count_);
486 self->impl_->transition_to(LinkState::Error);
487 return;
488 }
489
490 unsigned connection_timeout_ms;
491 {
492 std::lock_guard<std::mutex> lock(cfg_mtx_);
493 connection_timeout_ms = cfg_.connection_timeout_ms;
494 }
495 connect_timer_.expires_after(std::chrono::milliseconds(connection_timeout_ms));
496 connect_timer_.async_wait(net::bind_executor(strand_, [self, seq](const boost::system::error_code& ec) {
497 if (ec == net::error::operation_aborted || seq != self->impl_->current_seq_.load()) return;
498 if (!ec) {
499 self->impl_->record_error(diagnostics::ErrorLevel::ERROR, diagnostics::ErrorCategory::CONNECTION, "connect",
500 boost::asio::error::timed_out, "Connection timed out", true,
501 self->impl_->reconnect_attempt_count_);
502 self->impl_->handle_close(self, seq, boost::asio::error::timed_out);
503 }
504 }));
505
506 socket_->async_connect(*endpoint, net::bind_executor(strand_, [self, seq,
507 endpoint](const boost::system::error_code& ec) {
508 self->impl_->connect_timer_.cancel();
509 if (ec == net::error::operation_aborted || seq != self->impl_->current_seq_.load()) return;
510 if (self->impl_->stop_requested_.load() || self->impl_->stopping_.load()) {
511 self->impl_->close_socket();
512 return;
513 }
514 if (ec) {
515 self->impl_->record_error(diagnostics::ErrorLevel::ERROR, diagnostics::ErrorCategory::CONNECTION, "connect", ec,
516 "Connect failed: " + ec.message(), diagnostics::is_retryable_uds_connect_error(ec),
517 self->impl_->reconnect_attempt_count_);
518 self->impl_->schedule_retry(self, seq);
519 return;
520 }
521
522 self->impl_->connected_ = true;
523 self->impl_->reconnect_attempt_count_ = 0;
524 self->impl_->retry_attempts_ = 0;
525 self->impl_->transition_to(LinkState::Connected);
526 self->impl_->start_read(self, seq);
527 self->impl_->writing_ = false; // Force reset
528 self->impl_->do_write(self, seq);
529 }));
530}

532void UdsClient::Impl::schedule_retry(std::shared_ptr<UdsClient> self, uint64_t seq) {
533 transition_to(LinkState::Error);
534
535 // Snapshot once rather than locking repeatedly - cfg_/reconnect_policy_
536 // can change concurrently via set_retry_interval() etc. from any user
537 // thread while this runs on the strand (#436).
538 UdsClientConfig cfg_snapshot;
539 std::optional<ReconnectPolicy> reconnect_policy_snapshot;
540 {
541 std::lock_guard<std::mutex> lock(cfg_mtx_);
542 cfg_snapshot = cfg_;
543 reconnect_policy_snapshot = reconnect_policy_;
544 }
545
546 diagnostics::ErrorInfo dummy_err(diagnostics::ErrorLevel::ERROR, diagnostics::ErrorCategory::CONNECTION, "uds_client",
547 "connect", "Retry pending", boost::system::error_code(), true);
548 auto decision =
549 detail::decide_reconnect_uds(cfg_snapshot, dummy_err, reconnect_attempt_count_, reconnect_policy_snapshot);
550
551 if (!decision.should_retry || stop_requested_.load() || stopping_.load()) {
552 transition_to(LinkState::Idle);
553 return;
554 }
555
556 reconnect_attempt_count_++;
557 retry_timer_.expires_after(decision.delay.value_or(std::chrono::milliseconds(cfg_snapshot.retry_interval_ms)));
558 retry_timer_.async_wait(net::bind_executor(strand_, [self, seq](const boost::system::error_code& ec) {
559 if (ec == net::error::operation_aborted || seq != self->impl_->current_seq_.load()) return;
560 self->impl_->do_connect(self, seq);
561 }));
562}

564void UdsClient::Impl::start_read(std::shared_ptr<UdsClient> self, uint64_t seq) {
565 if (stop_requested_.load() || stopping_.load() || seq != current_seq_.load()) {
566 return;
567 }
568
569 socket_->async_read_some(net::buffer(rx_),
570 net::bind_executor(strand_, [self, seq](const boost::system::error_code& ec, size_t bytes) {
571 if (ec == net::error::operation_aborted || seq != self->impl_->current_seq_.load()) return;
572 if (self->impl_->stop_requested_.load() || self->impl_->stopping_.load()) return;
573 if (ec) {
574 self->impl_->handle_close(self, seq, ec);
575 return;
576 }
577
578 OnBytes cb;
579 {
580 std::lock_guard<std::mutex> lock(self->impl_->callback_mtx_);
581 cb = self->impl_->on_bytes_;
582 }
583 if (bytes > 0) self->impl_->stats_.record_received(bytes);
584 if (cb) cb(memory::ConstByteSpan(self->impl_->rx_.data(), bytes));
585 self->impl_->start_read(self, seq);
586 }));
587}

589void UdsClient::Impl::do_write(std::shared_ptr<UdsClient> self, uint64_t seq) {
590 if (stop_requested_.load() || stopping_.load() || seq != current_seq_.load()) {
591 tx_.clear();
592 pending_.clear();
593 queue_bytes_ = 0;
594 pending_bytes_ = 0;
595 current_write_buffer_ = std::nullopt;
596 writing_ = false;
597 return;
598 }
599
600 if (tx_.empty() || writing_) return;
601 writing_ = true;
602 current_write_buffer_ = std::move(tx_.front());
603 tx_.pop_front();
604
605 net::const_buffer buffer;
606 std::visit(
607 [&buffer](auto&& arg) {
608 using T = std::decay_t<decltype(arg)>;
609 if constexpr (std::is_same_v<T, std::vector<uint8_t>>)
610 buffer = net::buffer(arg);
611 else if constexpr (std::is_same_v<T, std::shared_ptr<const std::vector<uint8_t>>>)
612 buffer = net::buffer(*arg);
613 else if constexpr (std::is_same_v<T, memory::PooledBuffer>)
614 buffer = net::buffer(arg.data(), arg.size());
615 },
616 *current_write_buffer_);
617
618 size_t bytes_to_write = buffer.size();
619 socket_->async_write(
620 buffer,
621 net::bind_executor(strand_, [self, seq, bytes_to_write](const boost::system::error_code& ec, size_t written) {
622 if (ec == net::error::operation_aborted || seq != self->impl_->current_seq_.load()) return;
623 if (self->impl_->stop_requested_.load() || self->impl_->stopping_.load()) {
624 self->impl_->current_write_buffer_ = std::nullopt;
625 self->impl_->writing_ = false;
626 return;
627 }
628 self->impl_->writing_ = false;
629 self->impl_->current_write_buffer_ = std::nullopt;
630 self->impl_->queue_bytes_ =
631 (self->impl_->queue_bytes_ >= bytes_to_write) ? (self->impl_->queue_bytes_ - bytes_to_write) : 0;
632 self->impl_->report_backpressure(self, self->impl_->queue_bytes_);
633
634 if (ec) {
635 self->impl_->handle_close(self, seq, ec);
636 return;
637 }
638 self->impl_->stats_.record_sent(written);
639 if (!self->impl_->tx_.empty()) self->impl_->do_write(self, seq);
640 }));
641}

715void UdsClient::Impl::route_enqueued_buffer(std::shared_ptr<UdsClient> self, BufferVariant&& buf, size_t added) {
716 auto f = bp_fields();
717 queue_util::DropAccounting dropped;
718 auto decision = queue_util::decide_enqueue(f, added, tx_, dropped);
719 if (dropped.any()) stats_.record_dropped(dropped.messages, dropped.bytes);
720
721 if (decision == queue_util::EnqueueDecision::Rejected) {
722 UNILINK_LOG_ERROR("uds_client", "write", fmt::format("Queue limit exceeded ({} bytes)", queue_bytes_ + added));
723 report_backpressure(self, queue_bytes_ + added);
724 return;
725 }
726 if (decision == queue_util::EnqueueDecision::Pending) {
727 pending_bytes_ += added;
728 pending_.emplace_back(std::move(buf));
729 observe_queue();
730 return;
731 }
732 queue_bytes_ += added;
733 tx_.emplace_back(std::move(buf));
734 observe_queue();
735 report_backpressure(self, queue_bytes_);
736 if (!writing_) do_write(self, current_seq_.load());
737}

743void UdsClient::Impl::report_backpressure(std::shared_ptr<UdsClient> self, size_t queued_bytes) {
744 if (stop_requested_.load() || stopping_.load()) return;
745 observe_queue();
746
747 OnBackpressure on_bp;
748 {
749 std::lock_guard<std::mutex> lock(callback_mtx_);
750 on_bp = on_bp_;
751 }
752
753 auto f = bp_fields();
754 queue_util::report_backpressure(
755 f, queued_bytes, on_bp, stats_,
756 [&]() -> size_t {
757 const size_t moved = pending_bytes_.exchange(0);
758 while (!pending_.empty()) {
759 tx_.emplace_back(std::move(pending_.front()));
760 pending_.pop_front();
761 }
762 return moved;
763 },
764 [&]() {
765 observe_queue();
766 if (!writing_) do_write(self, current_seq_.load());
767 });
768}
