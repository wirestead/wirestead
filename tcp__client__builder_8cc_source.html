28template <uint32_t State>
29TcpClientBuilder<State>::TcpClientBuilder(const std::string& host, uint16_t port)
30 : host_(host),
31 port_(port),
32 auto_start_(false),
33 independent_context_(false),
34 retry_interval_(base::constants::DEFAULT_RETRY_INTERVAL_MS),
35 max_retries_(base::constants::DEFAULT_MAX_RETRIES),
36 connection_timeout_(base::constants::DEFAULT_CONNECTION_TIMEOUT_MS),
37 idle_timeout_(0),
38 idle_timeout_action_(IdleTimeoutAction::Reconnect),
39 tcp_no_delay_(true),
40 keep_alive_(false),
41 send_buffer_size_(0),
42 receive_buffer_size_(0) {
43 if (port == 0) throw diagnostics::BuilderException("Invalid port number: 0");
44 if (host.empty()) throw diagnostics::BuilderException("Host cannot be empty");

46 // Ensure background IO service is running
47 AutoInitializer::ensure_io_context_running();
48}

50template <uint32_t State>
51std::unique_ptr<wrapper::TcpClient> TcpClientBuilder<State>::build() {
52 std::unique_ptr<wrapper::TcpClient> client;
53 if (independent_context_) {
54 client = std::make_unique<wrapper::TcpClient>(host_, port_, std::make_shared<boost::asio::io_context>());
55 client->manage_external_context(true);
56 } else {
57 client = std::make_unique<wrapper::TcpClient>(host_, port_);
58 }
59
60 if (this->on_data_) client->on_data(this->on_data_);
61 if (this->on_data_batch_) client->on_data_batch(this->on_data_batch_);
62 if (this->on_connect_) client->on_connect(this->on_connect_);
63 if (this->on_disconnect_) client->on_disconnect(this->on_disconnect_);
64 if (this->on_error_) client->on_error(this->on_error_);
65 if (this->on_backpressure_) client->on_backpressure(this->on_backpressure_);
66
67 if (retry_interval_set_) client->retry_interval(retry_interval_);
68 if (max_retries_set_) client->max_retries(max_retries_);
69 if (connection_timeout_set_) client->connection_timeout(connection_timeout_);
70 if (idle_timeout_set_) client->idle_timeout(idle_timeout_);
71 if (idle_timeout_action_set_) client->idle_timeout_action(idle_timeout_action_);
72 if (tcp_no_delay_set_) client->tcp_no_delay(tcp_no_delay_);
73 if (keep_alive_set_) client->keep_alive(keep_alive_);
74 if (send_buffer_size_set_) client->send_buffer_size(send_buffer_size_);
75 if (receive_buffer_size_set_) client->receive_buffer_size(receive_buffer_size_);
76
77 if (this->bp_strategy_set_) client->backpressure_strategy(this->bp_strategy_);
78 client->backpressure_threshold(this->get_effective_backpressure_threshold());
79
80 if (this->framer_factory_) {
81 client->framer(this->framer_factory_());
82 }
83 if (this->on_message_) {
84 client->on_message(std::move(this->on_message_));
85 }
86 if (this->on_message_batch_) {
87 client->on_message_batch(std::move(this->on_message_batch_));
88 }
89
90 if (auto_start_) {
91 client->auto_start(true);
92 }
93
94 return client;
95}
