40// std::hash<boost::asio::ip::udp::endpoint> is not available before Boost 1.74.
41// Provide a portable hash by combining the raw address bytes and port.
42struct UdpEndpointHash {
43 std::size_t operator()(const boost::asio::ip::udp::endpoint& ep) const noexcept {
44 std::size_t seed = 0;
45 auto combine = [&](std::size_t v) { seed ^= v + 0x9e3779b9u + (seed << 6) + (seed >> 2); };
46 if (ep.address().is_v4()) {
47 for (auto byte : ep.address().to_v4().to_bytes()) {
48 combine(std::hash<unsigned char>{}(byte));
49 }
50 } else {
51 for (auto byte : ep.address().to_v6().to_bytes()) {
52 combine(std::hash<unsigned char>{}(byte));
53 }
54 }
55 combine(std::hash<unsigned short>{}(ep.port()));
56 return seed;
57 }
58};

127 void fulfill_all_locked(bool value) {
128 for (auto& promise : pending_promises) {
129 try {
130 promise.set_value(value);
131 } catch (...) {
132 }
133 }
134 pending_promises.clear();
135 }
136
137 void flush_batches() {
138 std::unique_lock<std::shared_mutex> lock(mutex);
139 if (!data_batch_queue_.empty()) {
140 auto handler = on_data_batch_;
141 auto batch = std::move(data_batch_queue_);
142 data_batch_queue_.clear();
143 if (handler) {
144 lock.unlock();
145 handler(batch);
146 lock.lock();
147 }
148 }
149 if (!message_batch_queue_.empty()) {
150 auto handler = on_message_batch_;
151 auto batch = std::move(message_batch_queue_);
152 message_batch_queue_.clear();
153 if (handler) {
154 lock.unlock();
155 handler(batch);
156 lock.lock();
157 }
158 }
159 if (batch_timer_) {
160 batch_timer_->cancel();
161 }
162 }
163
164 void schedule_batch_timer() {
165 if (!batch_timer_) return;
166 batch_timer_->expires_after(max_batch_latency_);
167 batch_timer_->async_wait([this, alive = std::weak_ptr<bool>(is_alive)](const boost::system::error_code& ec) {
168 auto lock = alive.lock();
169 if (!lock || !(*lock)) return;
170
171 if (!ec) {
172 flush_batches();
173 }
174 });
175 }
176
177 void schedule_reaper() {
178 if (!started.load() || !reaper_timer || session_timeout.count() <= 0) return;
179
180 // Run reaper at interval proportional to timeout (min 100ms, max 5s)
181 auto interval =
182 std::max(std::chrono::milliseconds(100), std::min(std::chrono::milliseconds(5000), session_timeout / 2));
183
184 reaper_timer->expires_after(interval);
185 reaper_timer->async_wait([this, alive = std::weak_ptr<bool>(is_alive)](const boost::system::error_code& ec) {
186 auto lock = alive.lock();
187 if (!lock || !(*lock)) return;
188
189 if (!ec) {
190 run_reaper();
191 schedule_reaper();
192 }
193 });
194 }
195
196 void run_reaper() {
197 if (session_timeout.count() <= 0) return;
198
199 std::vector<std::pair<ClientId, std::string>> to_remove_with_info;
200 auto now = std::chrono::steady_clock::now();
201
202 ConnectionHandler disconnect_handler;
203 {
204 std::unique_lock<std::shared_mutex> lock(mutex);
205 for (auto it = sessions.begin(); it != sessions.end();) {
206 if (now - it->second.last_seen > session_timeout) {
207 std::string info =
208 fmt::format("{}:{}", it->second.endpoint.address().to_string(), it->second.endpoint.port());
209 endpoint_to_id.erase(it->second.endpoint);
210 to_remove_with_info.push_back({it->first, info});
211 it = sessions.erase(it);
212 } else {
213 ++it;
214 }
215 }
216 disconnect_handler = on_disconnect;
217 }
218
219 // Call disconnect handlers outside the lock
220 if (disconnect_handler) {
221 for (auto const& [id, info] : to_remove_with_info) {
222 disconnect_handler(ConnectionContext(id, info));
223 }
224 }
225 }

227 void setup_internal_handlers() {
228 if (!channel) return;
229
230 batch_timer_ = std::make_unique<boost::asio::steady_timer>(channel->get_executor());
231
232 channel->on_bytes_from([this](memory::ConstByteSpan data, const boost::asio::ip::udp::endpoint& ep) {
233 ClientId client_id = 0;
234 bool is_new = false;
235 ConnectionHandler connect_handler_copy{nullptr};
236
237 {
238 std::unique_lock<std::shared_mutex> lock(mutex);
239 auto it = endpoint_to_id.find(ep);
240 if (it == endpoint_to_id.end()) {
241 if (client_limit_enabled.load() && sessions.size() >= max_clients_limit.load()) {
242 return;
243 }
244 client_id = next_client_id++;
245 endpoint_to_id[ep] = client_id;
246 SessionEntry entry;
247 entry.endpoint = ep;
248 entry.last_seen = std::chrono::steady_clock::now();
249 is_new = true;
250
251 // Create framer for new session
252 if (framer_factory) {
253 auto framer = framer_factory();
254 if (framer) {
255 framer->on_message([this, client_id](memory::ConstByteSpan msg) {
256 std::unique_lock<std::shared_mutex> lock(mutex);
257 if (on_message_batch_) {
258 message_batch_queue_.emplace_back(client_id, memory::SafeDataBuffer(msg));
259 if (message_batch_queue_.size() >= max_batch_size_) {
260 auto handler = on_message_batch_;
261 auto batch = std::move(message_batch_queue_);
262 message_batch_queue_.clear();
263 lock.unlock();
264 handler(batch);
265 } else if (message_batch_queue_.size() == 1) {
266 schedule_batch_timer();
267 }
268 return;
269 }
270
271 MessageHandler on_message_handler = on_message;
272 if (on_message_handler) {
273 lock.unlock();
274 on_message_handler(MessageContext(client_id, memory::SafeDataBuffer(msg)));
275 }
276 });
277 entry.framer = std::move(framer);
278 }
279 }
280 sessions[client_id] = std::move(entry);
281 } else {
282 client_id = it->second;
283 sessions[client_id].last_seen = std::chrono::steady_clock::now();
284 }
285 connect_handler_copy = on_connect;
286 }
287
288 if (is_new && connect_handler_copy) {
289 connect_handler_copy(ConnectionContext(client_id, fmt::format("{}:{}", ep.address().to_string(), ep.port())));
290 }
291
292 {
293 std::unique_lock<std::shared_mutex> lock(mutex);
294 if (on_data_batch_) {
295 data_batch_queue_.emplace_back(client_id, memory::SafeDataBuffer(data));
296 if (data_batch_queue_.size() >= max_batch_size_) {
297 auto handler = on_data_batch_;
298 auto batch = std::move(data_batch_queue_);
299 data_batch_queue_.clear();
300 lock.unlock();
301 handler(batch);
302 } else if (data_batch_queue_.size() == 1) {
303 schedule_batch_timer();
304 }
305 } else {
306 MessageHandler data_handler_copy = on_data;
307 if (data_handler_copy) {
308 lock.unlock();
309 data_handler_copy(MessageContext(client_id, memory::SafeDataBuffer(data)));
310 lock.lock();
311 }
312 }
313 }
314
315 // Push to framer
316 std::shared_ptr<framer::IFramer> target_framer;
317 {
318 std::shared_lock<std::shared_mutex> lock(mutex);
319 auto it = sessions.find(client_id);
320 if (it != sessions.end()) {
321 target_framer = it->second.framer;
322 }
323 }
324 if (target_framer) {
325 target_framer->push_bytes(data);
326 }
327 });
328
329 channel->on_backpressure([this](size_t queued) {
330 bp_cv_.notify_all();
331 std::function<void(size_t)> handler;
332 {
333 std::shared_lock<std::shared_mutex> lock(mutex);
334 handler = bp_handler;
335 }
336 if (handler) handler(queued);
337 });
338
339 channel->on_state([this](base::LinkState state) {
340 ErrorHandler error_handler_copy{nullptr};
341 if (state == base::LinkState::Listening || state == base::LinkState::Connected) {
342 is_listening.store(true);
343 std::unique_lock<std::shared_mutex> lock(mutex);
344 fulfill_all_locked(true);
345 } else if (state == base::LinkState::Error || state == base::LinkState::Closed ||
346 state == base::LinkState::Idle) {
347 is_listening.store(false);
348 std::unique_lock<std::shared_mutex> lock(mutex);
349 fulfill_all_locked(false);
350 if (state == base::LinkState::Error) {
351 error_handler_copy = on_error;
352 }
353 }
354
355 if (error_handler_copy) {
356 error_handler_copy(ErrorContext(ErrorCode::IoError, "Server error"));
357 }
358 });
359 }

361 std::future<bool> start() {
362 std::unique_lock<std::shared_mutex> lock(mutex);
363 if (is_listening.load()) {
364 std::promise<bool> p;
365 p.set_value(true);
366 return p.get_future();
367 }
368
369 std::promise<bool> p;
370 auto fut = p.get_future();
371 pending_promises.emplace_back(std::move(p));
372
373 if (started.exchange(true)) {
374 return fut;
375 }
376
377 if (!channel) {
378 channel = std::dynamic_pointer_cast<transport::UdpChannel>(factory::ChannelFactory::create(cfg, external_ioc));
379 setup_internal_handlers();
380 }
381
382 lock.unlock();
383 channel->start();
384
385 lock.lock();
386 if (use_external_context.load() && manage_external_context.load() && !external_thread.joinable()) {
387 if (external_ioc->stopped()) external_ioc->restart();
388 work_guard = std::make_unique<boost::asio::executor_work_guard<boost::asio::io_context::executor_type>>(
389 external_ioc->get_executor());
390 external_thread = std::jthread([ioc = external_ioc](std::stop_token st) {
391 try {
392 std::stop_callback cb(st, [ioc] { ioc->stop(); });
393 ioc->run();
394 } catch (...) {
395 }
396 });
397 }
398
399 if (channel && session_timeout.count() > 0) {
400 reaper_timer = std::make_unique<boost::asio::steady_timer>(channel->get_executor());
401 schedule_reaper();
402 }
403
404 return fut;
405 }

407 void stop() {
408 bool should_join = false;
409 {
410 std::unique_lock<std::shared_mutex> lock(mutex);
411 if (!started.exchange(false)) {
412 is_listening.store(false);
413 fulfill_all_locked(false);
414 return;
415 }
416 bp_cv_.notify_all();
417
418 if (reaper_timer) {
419 reaper_timer->cancel();
420 reaper_timer.reset();
421 }
422
423 if (batch_timer_) {
424 batch_timer_->cancel();
425 batch_timer_.reset();
426 }
427
428 if (channel) {
429 lock.unlock();
430 channel->stop();
431 // Clear callbacks after stop() rather than before: the
432 // transport-level fix (#436) already synchronizes callback reads
433 // against these setters, but clearing after stop() means no
434 // in-flight handler can observe a null callback mid-shutdown in the
435 // first place - belt and braces once the underlying race is fixed
436 // at the source. Also now clears on_backpressure, which this path
437 // previously never did at all.
438 channel->on_bytes_from(nullptr);
439 channel->on_state(nullptr);
440 channel->on_backpressure(nullptr);
441 lock.lock();
442 }
443
444 if (use_external_context.load() && manage_external_context.load()) {
445 if (external_ioc) external_ioc->stop();
446 should_join = true;
447 }
448
449 is_listening.store(false);
450 endpoint_to_id.clear();
451 sessions.clear();
452 next_client_id = 1;
453 fulfill_all_locked(false);
454 }
455
456 if (should_join && external_thread.joinable()) {
457 try {
458 if (std::this_thread::get_id() != external_thread.get_id()) {
459 external_thread.request_stop();
460 external_thread.join();
461 } else {
462 external_thread.detach();
463 }
464 } catch (...) {
465 }
466 }
467 std::unique_lock<std::shared_mutex> lock(mutex);
468 channel.reset();
469 }

471 bool send_to(ClientId client_id, std::string_view data) {
472 if (cfg.backpressure_strategy == base::constants::BackpressureStrategy::Reliable)
473 return send_to_blocking(client_id, data);
474 return try_send_to(client_id, data);
475 }
476
477 bool try_broadcast(std::string_view data) {
478 std::shared_lock<std::shared_mutex> lock(mutex);
479 if (!channel) return false;
480 bool sent = false;
481 auto bytes = base::safe_convert::string_to_bytes(data);
482 for (const auto& [id, entry] : sessions) {
483 sent |= channel->async_try_write_to(memory::ConstByteSpan(bytes.first, bytes.second), entry.endpoint);
484 }
485 return sent;
486 }
487
488 bool broadcast(std::string_view data) { return try_broadcast(data); }
489
490 // channel->on_backpressure() calls bp_cv_.notify_all() from the transport's io_context
491 // thread without holding bp_mutex_ (backpressure_active_ is a plain atomic on the transport
492 // side, not guarded by bp_mutex_ at all). That makes a classic lost-wakeup race possible: a
493 // waiter can check the predicate, find it still blocking, and be in the process of
494 // registering to wait when the notify fires - in the rare case that race is lost, an
495 // unbounded wait() would block forever. Poll with a bounded timeout instead so a missed
496 // notify only costs a short delay rather than a permanent hang (see #427, #431).
497 bool send_to_blocking(ClientId client_id, std::string_view data) {
498 std::unique_lock<std::mutex> bp_lock(bp_mutex_);
499 auto predicate = [this] {
500 std::shared_lock<std::shared_mutex> lock(mutex);
501 return !started.load() || !channel || !channel->is_backpressure_active();
502 };
503 while (!bp_cv_.wait_for(bp_lock, std::chrono::milliseconds(50), predicate)) {
504 }
505 return try_send_to(client_id, data);
506 }
507
508 bool try_send_to(ClientId client_id, std::string_view data) {
509 std::shared_lock<std::shared_mutex> lock(mutex);
510 auto it = sessions.find(client_id);
511 if (it == sessions.end() || !channel) return false;
512
513 auto bytes = base::safe_convert::string_to_bytes(data);
514 return channel->async_try_write_to(memory::ConstByteSpan(bytes.first, bytes.second), it->second.endpoint);
515 }

518 std::shared_lock<std::shared_mutex> lock(mutex);
519 return channel ? channel->stats() : RuntimeStats{};
520 }
521
522 void reset_stats() {
523 std::shared_lock<std::shared_mutex> lock(mutex);
524 if (channel) channel->reset_stats();
525 }

670UdpServer& UdpServer::max_clients(size_t max) {
