114 void fulfill_all_locked(bool value) {
115 for (auto& promise : pending_promises_) {
116 try {
117 promise.set_value(value);
118 } catch (...) {
119 }
120 }
121 pending_promises_.clear();
122 }

124 void flush_batches() {
125 std::unique_lock<std::shared_mutex> lock(mutex_);
126 if (!data_batch_queue_.empty()) {
127 auto handler = data_batch_handler_;
128 auto batch = std::move(data_batch_queue_);
129 data_batch_queue_.clear();
130 if (handler) {
131 lock.unlock();
132 handler(batch);
133 lock.lock();
134 }
135 }
136 if (!message_batch_queue_.empty()) {
137 auto handler = message_batch_handler_;
138 auto batch = std::move(message_batch_queue_);
139 message_batch_queue_.clear();
140 if (handler) {
141 lock.unlock();
142 handler(batch);
143 lock.lock();
144 }
145 }
146 if (batch_timer_) {
147 batch_timer_->cancel();
148 }
149 }
150
151 void schedule_batch_timer() {
152 if (!batch_timer_) return;
153 batch_timer_->expires_after(max_batch_latency_);
154 batch_timer_->async_wait(
155 [this, weak_alive = std::weak_ptr<bool>(alive_marker_)](const boost::system::error_code& ec) {
156 if (ec) return;
157 auto alive = weak_alive.lock();
158 if (!alive) return;
159 flush_batches();
160 });
161 }

163 std::future<bool> start() {
164 std::unique_lock<std::shared_mutex> lock(mutex_);
165 if (channel && channel->is_connected()) {
166 started_.store(true);
167 std::promise<bool> p;
168 p.set_value(true);
169 return p.get_future();
170 }
171
172 std::promise<bool> p;
173 auto future = p.get_future();
174 pending_promises_.emplace_back(std::move(p));
175
176 if (started_.load()) {
177 return future;
178 }
179
180 if (!channel) {
181 channel = factory::ChannelFactory::create(build_config_locked(), external_ioc);
182 setup_internal_handlers();
183 }
184 started_.store(true);
185
186 lock.unlock();
187 channel->start();
188 if (use_external_context && manage_external_context && !external_thread.joinable()) {
189 if (external_ioc && external_ioc->stopped()) {
190 external_ioc->restart();
191 }
192 work_guard_ = std::make_unique<boost::asio::executor_work_guard<boost::asio::io_context::executor_type>>(
193 boost::asio::make_work_guard(*external_ioc));
194 external_thread = std::thread([ioc = external_ioc]() {
195 try {
196 ioc->run();
197 } catch (...) {
198 }
199 });
200 }
201
202 return future;
203 }

205 void stop() {
206 std::unique_lock<std::shared_mutex> lock(mutex_);
207 if (!started_.load()) {
208 fulfill_all_locked(false);
209 return;
210 }
211 started_.store(false);
212 bp_cv_.notify_all();
213
214 if (batch_timer_) {
215 batch_timer_->cancel();
216 batch_timer_.reset();
217 }
218
219 if (channel) {
220 lock.unlock();
221 channel->stop();
222 // Clear callbacks after stop() rather than before: the transport-level
223 // fix (#436) already synchronizes callback reads against these
224 // setters, but clearing after stop() means no in-flight handler can
225 // observe a null callback mid-shutdown in the first place - belt and
226 // braces once the underlying race is fixed at the source.
227 channel->on_bytes(nullptr);
228 channel->on_state(nullptr);
229 channel->on_backpressure(nullptr);
230 lock.lock();
231 if (factory_managed_channel_) {
232 // Fully release the channel rather than reusing it: start()'s
233 // `if (!channel)` guard is what re-runs setup_internal_handlers() and
234 // rebuilds config from the (possibly changed) staged fields. Reusing
235 // a stopped channel left every handler nulled forever - including
236 // the one that fulfills the start() future - so a restart would
237 // hang (jwsung91/unilink#444). Injected channels (factory_managed_
238 // channel_ == false) are exempt: the caller owns that channel's
239 // identity, so we must not discard and factory-rebuild it.
240 channel.reset();
241 }
242 }
243
244 if (work_guard_) {
245 work_guard_.reset();
246 }
247
248 if (use_external_context && manage_external_context && external_thread.joinable()) {
249 if (external_ioc) external_ioc->stop();
250 if (std::this_thread::get_id() != external_thread.get_id()) {
251 lock.unlock();
252 external_thread.join();
253 lock.lock();
254 } else {
255 external_thread.detach();
256 }
257 }
258
259 fulfill_all_locked(false);
260
261 if (framer) framer->reset();
262 }

292 return try_send(data);

294
295 // channel->on_backpressure() calls bp_cv_.notify_all() from the transport's io_context
296 // thread without holding bp_mutex_ (backpressure_active_ is a plain atomic on the transport
297 // side, not guarded by bp_mutex_ at all). That makes a classic lost-wakeup race possible: a
298 // waiter can check the predicate, find it still blocking, and be in the process of
299 // registering to wait when the notify fires - in the rare case that race is lost, an
300 // unbounded wait() would block forever. Poll with a bounded timeout instead so a missed
301 // notify only costs a short delay rather than a permanent hang (see #427, #431).
302 void wait_for_backpressure_clear(std::unique_lock<std::mutex>& bp_lock) {
303 auto predicate = [this] {
304 std::shared_lock<std::shared_mutex> lock(mutex_);
305 return !started_.load() || !channel || !channel->is_connected() || !channel->is_backpressure_active();
306 };
307 while (!bp_cv_.wait_for(bp_lock, std::chrono::milliseconds(50), predicate)) {
308 }
309 }

311 bool send_move(std::vector<uint8_t>&& data) {
312 if (backpressure_strategy == base::constants::BackpressureStrategy::Reliable) {
313 std::unique_lock<std::mutex> bp_lock(bp_mutex_);
314 wait_for_backpressure_clear(bp_lock);
315 std::shared_lock<std::shared_mutex> lock(mutex_);
316 if (!started_.load() || !channel || !channel->is_connected()) return false;
317 return channel->async_write_move(std::move(data));
318 }
319 return try_send_move(std::move(data));
320 }

325 std::unique_lock<std::mutex> bp_lock(bp_mutex_);

332 }

363 void setup_internal_handlers() {
364 if (!channel) return;
365
366 batch_timer_ = std::make_unique<boost::asio::steady_timer>(channel->get_executor());
367
368 std::weak_ptr<bool> weak_alive = alive_marker_;
369
370 channel->on_bytes([this, weak_alive](memory::ConstByteSpan data) {
371 auto alive = weak_alive.lock();
372 if (!alive) return;
373
374 std::shared_ptr<framer::IFramer> framer_to_push;
375 std::unique_lock<std::shared_mutex> lock(mutex_);
376 if (data_batch_handler_) {
377 data_batch_queue_.emplace_back(0, memory::SafeDataBuffer(data));
378 if (data_batch_queue_.size() >= max_batch_size_) {
379 auto handler = data_batch_handler_;
380 auto batch = std::move(data_batch_queue_);
381 data_batch_queue_.clear();
382 lock.unlock();
383 handler(batch);
384 lock.lock();
385 } else if (data_batch_queue_.size() == 1) {
386 schedule_batch_timer();
387 }
388 } else {
389 MessageHandler handler = data_handler;
390 if (handler) {
391 lock.unlock();
392 handler(MessageContext(0, memory::SafeDataBuffer(data)));
393 lock.lock();
394 }
395 }
396
397 if (framer) {
398 framer_to_push = framer;
399 }
400 lock.unlock();
401 if (framer_to_push) framer_to_push->push_bytes(data);
402 });
403
404 channel->on_backpressure([this, weak_alive](size_t queued) {
405 bp_cv_.notify_all();
406 auto alive = weak_alive.lock();
407 if (!alive) return;
408 std::function<void(size_t)> handler;
409 {
410 std::shared_lock<std::shared_mutex> lock(mutex_);
411 handler = bp_handler;
412 }
413 if (handler) handler(queued);
414 });
415
416 channel->on_state([this, weak_alive](base::LinkState state) {
417 auto alive = weak_alive.lock();
418 if (!alive) return;
419
420 switch (state) {
421 case base::LinkState::Connected: {
422 ConnectionHandler handler;
423 {
424 std::unique_lock<std::shared_mutex> lock(mutex_);
425 fulfill_all_locked(true);
426 handler = connect_handler;
427 }
428 if (handler) handler(ConnectionContext(0));
429 break;
430 }
431 case base::LinkState::Closed:
432 case base::LinkState::Error:
433 case base::LinkState::Idle: {
434 ConnectionHandler disconnect_handler_snapshot;
435 ErrorHandler error_handler_snapshot;
436 {
437 std::unique_lock<std::shared_mutex> lock(mutex_);
438 fulfill_all_locked(false);
439 if (state == base::LinkState::Error) {
440 error_handler_snapshot = error_handler;
441 } else {
442 disconnect_handler_snapshot = disconnect_handler;
443 }
444 }
445 if (disconnect_handler_snapshot) {
446 disconnect_handler_snapshot(ConnectionContext(0));
447 }
448 if (error_handler_snapshot) {
449 error_handler_snapshot(ErrorContext(ErrorCode::IoError, "Connection error"));
450 }
451 break;
452 }
453 default:
454 break;
455 }
456 });
457 }

459 void attach_framer_callback() {
460 if (!framer) return;
461 framer->on_message([this](memory::ConstByteSpan msg) {
462 std::unique_lock<std::shared_mutex> lock(mutex_);
463 if (message_batch_handler_) {
464 message_batch_queue_.emplace_back(0, memory::SafeDataBuffer(msg));
465 if (message_batch_queue_.size() >= max_batch_size_) {
466 auto handler = message_batch_handler_;
467 auto batch = std::move(message_batch_queue_);
468 message_batch_queue_.clear();
469 lock.unlock();
470 handler(batch);
471 } else if (message_batch_queue_.size() == 1) {
472 schedule_batch_timer();
473 }
474 return;
475 }
476
477 MessageHandler handler = message_handler;
478 if (handler) {
479 lock.unlock();
480 handler(MessageContext(0, memory::SafeDataBuffer(msg)));
481 }
482 });
483 }

487 framer = std::shared_ptr<framer::IFramer>(std::move(f));
488 if (framer && (message_handler || message_batch_handler_)) attach_framer_callback();
489 }

503 config::SerialConfig build_config_locked() const {
504 config::SerialConfig config;
505 config.device = device;
506 config.baud_rate = baud_rate;
507 config.char_size = static_cast<unsigned int>(data_bits);
508 config.stop_bits = static_cast<unsigned int>(stop_bits);
509 std::string p = to_lower(parity);
510 if (p == "even")
511 config.parity = config::SerialConfig::Parity::Even;
512 else if (p == "odd")
513 config.parity = config::SerialConfig::Parity::Odd;
514 else
515 config.parity = config::SerialConfig::Parity::None;
516
517 std::string f = to_lower(flow_control);
518 if (f == "software")
519 config.flow = config::SerialConfig::Flow::Software;
520 else if (f == "hardware")
521 config.flow = config::SerialConfig::Flow::Hardware;
522 else
523 config.flow = config::SerialConfig::Flow::None;
524
525 config.retry_interval_ms = static_cast<unsigned int>(retry_interval.count());
526 config.reopen_on_error = reopen_on_error;
527 config.backpressure_threshold = backpressure_threshold;
528 config.backpressure_strategy = backpressure_strategy;
529 return config;
530 }
