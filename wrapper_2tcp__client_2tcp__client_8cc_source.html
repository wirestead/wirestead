114 void fulfill_all_locked(bool value) {
115 for (auto& p : pending_promises_) {
116 try {
117 p.set_value(value);
118 } catch (...) {
119 }
120 }
121 pending_promises_.clear();
122 }
123
124 void flush_batches() {
125 std::unique_lock<std::shared_mutex> lock(mutex_);
126 if (!data_batch_queue_.empty()) {
127 auto handler = data_batch_handler_;
128 auto batch = std::move(data_batch_queue_);
129 data_batch_queue_.clear();
130 if (handler) {
131 lock.unlock();
132 handler(batch);
133 lock.lock();
134 }
135 }
136 if (!message_batch_queue_.empty()) {
137 auto handler = message_batch_handler_;
138 auto batch = std::move(message_batch_queue_);
139 message_batch_queue_.clear();
140 if (handler) {
141 lock.unlock();
142 handler(batch);
143 lock.lock();
144 }
145 }
146 if (batch_timer_) {
147 batch_timer_->cancel();
148 }
149 }
150
151 void schedule_batch_timer() {
152 if (!batch_timer_) return;
153 batch_timer_->expires_after(max_batch_latency_);
154 batch_timer_->async_wait(
155 [this, weak_alive = std::weak_ptr<bool>(alive_marker_)](const boost::system::error_code& ec) {
156 if (ec) return;
157 auto alive = weak_alive.lock();
158 if (!alive) return;
159 flush_batches();
160 });
161 }

163 std::future<bool> start() {
164 std::unique_lock<std::shared_mutex> lock(mutex_);
165 if (channel_ && channel_->is_connected()) {
166 started_.store(true);
167 std::promise<bool> p;
168 p.set_value(true);
169 return p.get_future();
170 }
171 std::promise<bool> p;
172 auto f = p.get_future();
173 pending_promises_.push_back(std::move(p));
174 if (started_.load()) return f;
175
176 if (!alive_marker_) {
177 alive_marker_ = std::make_shared<bool>(true);
178 }
179
180 if (!channel_) {
181 config::TcpClientConfig config;
182 config.host = host_;
183 config.port = port_;
184 config.retry_interval_ms = static_cast<unsigned int>(retry_interval_.count());
185 config.max_retries = max_retries_;
186 config.connection_timeout_ms = static_cast<unsigned>(connection_timeout_.count());
187 config.idle_timeout_ms = static_cast<unsigned>(idle_timeout_.count());
188 config.idle_timeout_action = idle_timeout_action_;
189 config.backpressure_threshold = backpressure_threshold_;
190 config.backpressure_strategy = backpressure_strategy_;
191 config.tcp_no_delay = tcp_no_delay_;
192 config.keep_alive = keep_alive_;
193 config.send_buffer_size = send_buffer_size_;
194 config.receive_buffer_size = receive_buffer_size_;
195 channel_ = factory::ChannelFactory::create(config, external_ioc_);
196 setup_internal_handlers();
197 }
198
199 started_.store(true);
200 channel_->start();
201 if (use_external_context_.load() && manage_external_context_.load() && !external_thread_.joinable()) {
202 if (external_ioc_ && external_ioc_->stopped()) {
203 external_ioc_->restart();
204 }
205 work_guard_ = std::make_unique<boost::asio::executor_work_guard<boost::asio::io_context::executor_type>>(
206 external_ioc_->get_executor());
207 external_thread_ = std::jthread([this, ioc = external_ioc_](std::stop_token st) {
208 try {
209 while (!st.stop_requested() && started_.load() && !ioc->stopped()) {
210 if (ioc->run_one_for(std::chrono::milliseconds(50)) == 0) {
211 std::this_thread::yield();
212 }
213 }
214 } catch (...) {
215 }
216 });
217 }
218 return f;
219 }

221 void stop() {
222 bool should_join = false;
223 {
224 std::unique_lock<std::shared_mutex> lock(mutex_);
225 if (!started_.load()) {
226 fulfill_all_locked(false);
227 return;
228 }
229 started_.store(false);
230 bp_cv_.notify_all();
231 alive_marker_.reset();
232 if (batch_timer_) {
233 batch_timer_->cancel();
234 batch_timer_.reset();
235 }
236 if (channel_) {
237 auto ch = channel_;
238 lock.unlock();
239 ch->stop();
240 lock.lock();
241 if (channel_ == ch) {
242 channel_->on_bytes(nullptr);
243 channel_->on_state(nullptr);
244 }
245 }
246 if (use_external_context_.load() && manage_external_context_.load()) {
247 if (work_guard_) work_guard_.reset();
248 if (external_ioc_) external_ioc_->stop();
249 should_join = true;
250 }
251 fulfill_all_locked(false);
252 }
253 if (should_join && external_thread_.joinable()) {
254 try {
255 if (std::this_thread::get_id() != external_thread_.get_id()) {
256 external_thread_.request_stop();
257 external_thread_.join();
258 } else {
259 external_thread_.detach();
260 }
261 } catch (...) {
262 }
263 }
264 std::unique_lock<std::shared_mutex> lock(mutex_);
265 channel_.reset();
266 if (framer_) framer_->reset();
267 }

295 bool send(std::string_view data) {
296 if (backpressure_strategy_ == base::constants::BackpressureStrategy::Reliable) return send_blocking(data);
297 return try_send(data);
298 }
299
300 // channel_->on_backpressure() calls bp_cv_.notify_all() from the transport's io_context
301 // thread without holding bp_mutex_ (backpressure_active_ is a plain atomic on the transport
302 // side, not guarded by bp_mutex_ at all). That makes a classic lost-wakeup race possible: a
303 // waiter can check the predicate, find it still blocking, and be in the process of
304 // registering to wait when the notify fires - in the rare case that race is lost, an
305 // unbounded wait() would block forever. Poll with a bounded timeout instead so a missed
306 // notify only costs a short delay rather than a permanent hang (see #427, #431).
307 void wait_for_backpressure_clear(std::unique_lock<std::mutex>& bp_lock) {
308 auto predicate = [this] {
309 std::shared_lock<std::shared_mutex> lock(mutex_);
310 return !started_.load() || !channel_ || !channel_->is_backpressure_active();
311 };
312 while (!bp_cv_.wait_for(bp_lock, std::chrono::milliseconds(50), predicate)) {
313 }
314 }
315
316 bool send_move(std::vector<uint8_t>&& data) {
317 if (backpressure_strategy_ == base::constants::BackpressureStrategy::Reliable) {
318 std::unique_lock<std::mutex> bp_lock(bp_mutex_);
319 wait_for_backpressure_clear(bp_lock);
320 std::shared_lock<std::shared_mutex> lock(mutex_);
321 if (!started_.load() || !channel_ || !channel_->is_connected()) return false;
322 return channel_->async_write_move(std::move(data));
323 }
324 return try_send_move(std::move(data));
325 }
326
327 bool send_shared(std::shared_ptr<const std::vector<uint8_t>> data) {
328 if (!data || data->empty()) return false;
329 if (backpressure_strategy_ == base::constants::BackpressureStrategy::Reliable) {
330 std::unique_lock<std::mutex> bp_lock(bp_mutex_);
331 wait_for_backpressure_clear(bp_lock);
332 std::shared_lock<std::shared_mutex> lock(mutex_);
333 if (!started_.load() || !channel_ || !channel_->is_connected()) return false;
334 return channel_->async_write_shared(std::move(data));
335 }
336 return try_send_shared(std::move(data));
337 }

372 void setup_internal_handlers() {
373 if (!channel_) return;
374
375 batch_timer_ = std::make_unique<boost::asio::steady_timer>(channel_->get_executor());
376
377 std::weak_ptr<bool> weak_alive = alive_marker_;
378
379 channel_->on_bytes([this, weak_alive](memory::ConstByteSpan data) {
380 auto alive = weak_alive.lock();
381 if (!alive) return;
382
383 std::shared_ptr<framer::IFramer> framer_to_push;
384 std::unique_lock<std::shared_mutex> lock(mutex_);
385 if (data_batch_handler_) {
386 data_batch_queue_.emplace_back(0, memory::SafeDataBuffer(data));
387 if (data_batch_queue_.size() >= max_batch_size_) {
388 auto handler = data_batch_handler_;
389 auto batch = std::move(data_batch_queue_);
390 data_batch_queue_.clear();
391 lock.unlock();
392 handler(batch);
393 lock.lock();
394 } else if (data_batch_queue_.size() == 1) {
395 schedule_batch_timer();
396 }
397 } else {
398 MessageHandler handler = data_handler_;
399 if (handler) {
400 lock.unlock();
401 handler(MessageContext(0, memory::SafeDataBuffer(data)));
402 lock.lock();
403 }
404 }
405
406 if (framer_) {
407 framer_to_push = framer_;
408 }
409 lock.unlock();
410 if (framer_to_push) framer_to_push->push_bytes(data);
411 });
412
413 channel_->on_state([this, weak_alive](base::LinkState state) {
414 auto alive = weak_alive.lock();
415 if (!alive) return;
416 ConnectionHandler connect_handler;
417 ConnectionHandler disconnect_handler;
418 ErrorHandler error_handler;
419 std::shared_ptr<interface::Channel> channel_snapshot;
420
421 if (state == base::LinkState::Connected) {
422 {
423 std::unique_lock<std::shared_mutex> lock(mutex_);
424 fulfill_all_locked(true);
425 connect_handler = connect_handler_;
426 }
427 if (connect_handler) connect_handler(ConnectionContext(0));
428 } else if (state == base::LinkState::Closed || state == base::LinkState::Error) {
429 {
430 std::unique_lock<std::shared_mutex> lock(mutex_);
431 fulfill_all_locked(false);
432 if (state == base::LinkState::Closed) {
433 disconnect_handler = disconnect_handler_;
434 } else {
435 error_handler = error_handler_;
436 channel_snapshot = channel_;
437 }
438 }
439 if (state == base::LinkState::Closed && disconnect_handler) {
440 disconnect_handler(ConnectionContext(0));
441 } else if (state == base::LinkState::Error && error_handler) {
442 bool handled = false;
443 if (auto transport = std::dynamic_pointer_cast<transport::TcpClient>(channel_snapshot)) {
444 if (auto info = transport->last_error_info()) {
445 error_handler(diagnostics::to_error_context(*info));
446 handled = true;
447 }
448 }
449 if (!handled) {
450 error_handler(ErrorContext(ErrorCode::IoError, "Connection error"));
451 }
452 }
453 }
454 });
455
456 channel_->on_backpressure([this, weak_alive](size_t queued) {
457 bp_cv_.notify_all();
458 auto alive = weak_alive.lock();
459 if (!alive) return;
460 std::function<void(size_t)> handler;
461 {
462 std::shared_lock<std::shared_mutex> lock(mutex_);
463 handler = bp_handler_;
464 }
465 if (handler) handler(queued);
466 });
467 }
468
469 void attach_framer_callback() {
470 if (!framer_) return;
471 framer_->on_message([this](memory::ConstByteSpan msg) {
472 std::unique_lock<std::shared_mutex> lock(mutex_);
473 if (message_batch_handler_) {
474 message_batch_queue_.emplace_back(0, memory::SafeDataBuffer(msg));
475 if (message_batch_queue_.size() >= max_batch_size_) {
476 auto handler = message_batch_handler_;
477 auto batch = std::move(message_batch_queue_);
478 message_batch_queue_.clear();
479 lock.unlock();
480 handler(batch);
481 } else if (message_batch_queue_.size() == 1) {
482 schedule_batch_timer();
483 }
484 return;
485 }
486
487 MessageHandler handler = message_handler_;
488 if (handler) {
489 lock.unlock();
490 handler(MessageContext(0, memory::SafeDataBuffer(msg)));
491 }
492 });
493 }
494
495 void set_framer(std::unique_ptr<framer::IFramer> framer) {
496 std::unique_lock<std::shared_mutex> lock(mutex_);
497 framer_ = std::shared_ptr<framer::IFramer>(std::move(framer));
498 if (framer_ && (message_handler_ || message_batch_handler_)) attach_framer_callback();
499 }
500
501 void on_message(MessageHandler handler) {
502 std::unique_lock<std::shared_mutex> lock(mutex_);
503 message_handler_ = std::move(handler);
504 if (framer_) attach_framer_callback();
505 }
506
507 void on_message_batch(BatchMessageHandler handler) {
508 std::unique_lock<std::shared_mutex> lock(mutex_);
509 message_batch_handler_ = std::move(handler);
510 if (framer_) attach_framer_callback();
511 }

635TcpClient& TcpClient::idle_timeout(std::chrono::milliseconds t) {
636 std::unique_lock<std::shared_mutex> lock(impl_->mutex_);
